urls

Create a my_app/urls.py file in your app with the following patterns.

from django.conf import settings
from django.conf.urls.static import static
from django.urls import path
from myapp.views import (
    MyAppCreateView,
    MyAppUpdateView,
    MyAppDeleteView,
    MyAppEngageView,
    MyAppListView,
    MyAppIterateView,
    MyAppOptimizeView,
    MyAppPageView,
    MyAppHomeView,
)
urlpatterns = [
    path(
        "cr/<str:page_key>/<str:thing>",
        MyAppCreateView.as_view(),
        name="create.thing",
    ),
    path(
        "u/<str:page_key>/<str:thing>/<int:pk>",
        MyAppUpdateView.as_view(),
        name="update.thing",
    ),
    path(
        "d/<str:page_key>/<str:thing>/<int:pk>",
        MyAppDeleteView.as_view(),
        name="delete.thing",
    ),
    path(
        "e/<str:page_key>/<str:thing>/<int:pk>",
        MyAppEngageView.as_view(),
        name="engage.thing",
    ),
    path(
        "l/<str:page_key>/<str:thing>",
        MyAppListView.as_view(),
        name="list.things",
    ),
    path(
        "i/<str:page_key>/<str:thing>/<str:field>",
        MyAppIterateView.as_view(),
        name="iterate.things",
    ),
    path(
        "o/<str:page_key>/<str:thing>",
        MyAppOptimizeView.as_view(),
        name="optimize.things",
    ),
    path("p/<str:page_key>", MyAppPageView.as_view(), name="engage.page"),
    path("", MyAppHomeView.as_view(), name="home"),
]
if settings.DEBUG:
    urlpatterns += static(settings.MEDIA_URL, document_root=settings.MEDIA_ROOT)
